		
		#Region "Association <association_name>"

		friend _<association_name>Loaded as Boolean

		''' <summary>
        ''' Gets/Sets parent object
        ''' </summary>
		Public Overridable Property <association_name>() As <datatype> _
			implements <implements>.<association_name>
		    '1-1 parent association
            Set(ByVal value As <datatype>)
                Me._<association_name> = value
				If value isnot Nothing then
					me.<child_field_runtime> = value.<parent_field_runtime>
					AddHandler value.IDChanged, AddressOf Me.handleParentIdChanged
                Else
					me.<child_field_runtime> = Nothing
				End If

            End Set


            Get
                'LAZY LOADING! Only hit the database to get the child object if we need it
                If Me._<association_name> Is Nothing Then
					Me.load<association_name>
                End If 
				
                Return Me._<association_name>
            End Get
        End Property
        
        ''' <summary>
        ''' Loads parent object and sets the appropriate properties
        ''' </summary>
        Public Overridable Sub load<association_name>
			
			If me._<association_name>Loaded then return
			
			If Me._<association_name> Is Nothing AndAlso _
                   Me.<child_field_runtime> > 0 Then
                
				'call the setter here, not the private variable!
                Me.<association_name> = new <db_mapper>().FindByKey( _ 
				                 Me.<child_field_runtime>.Value)
                
            End If ' me._<association_name> is Nothing

            me._<association_name>Loaded=true
			            
        End Sub

		#End Region
